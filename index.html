        // Funcția pentru Ceas (Logic)

        function updateClock() {

        }

        setInterval(updateClock, 1000);

        updateClock();


        // Funcția pentru To-Do List (Interactivitate)

        function addTask() {

            let input = document.getElementById('taskInput');

            if (input.value === '') return;

            let li = document.createElement('li');

            li.innerHTML = `${input.value} <span style="cursor:pointer;color:#f87171" onclick="this.parentElement.remove()">✖</span>`;

            document.getElementById('taskList').appendChild(li);

            input.value = '';

        }
